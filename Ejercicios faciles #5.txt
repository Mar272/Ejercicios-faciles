#include <iostream>

using namespace std;
int calcularFactorial(int num) {
    if (num == 0 || num == 1) {
        return 1;
    } else {
        int factorial = 1;
        for (int i = 2; i <= num; i++) {
            factorial *= i;
        }
        return factorial;
    }
}

int main() {
    int num;

    cout << "Ingrese un nÃºmero: ";
    cin >> num;
    int factorial = calcularFactorial(num);

    cout << "El factorial de " << num << " es: " << factorial;

    return 0;
}
